\hypersetup{
    pdftitle={\titletext},
    pdfauthor={\authortext},
    colorlinks=false, linkcolor=black, citecolor=black, urlcolor=black
}

\newtoggle{usedcites}
\pretocmd{\cite}{\global\toggletrue{usedcites}}{}{}
\pretocmd{\citet}{\global\toggletrue{usedcites}}{}{}
\pretocmd{\citep}{\global\toggletrue{usedcites}}{}{}

\AtEndDocument{
    \iftoggle{usedcites}{
        \newpage
        \bibliographystyle{apacite}
        \bibliography{references}
    }
}
