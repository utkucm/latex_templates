\newcommand{\maketitlepage}{%
  \begin{titlepage}
    \thispagestyle{empty}
    \centering
    \vspace*{4cm}
    {\Huge \textbf{\coursename}}\par
    \vspace{0.5cm}
    {\LARGE \textbf{\assignmenttitle}}\par
    \vfill
    {\Large \studentname}\par
    {\large ID\@: \studentid}\par % chktex 13
    \vspace{0.3cm}
    {\large \duedate}\par
    \vspace*{2cm}
  \end{titlepage}%
}
