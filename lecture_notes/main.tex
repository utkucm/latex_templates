\newcommand{\newlecture}[4]{%
    \setcounter{lecture}{#1}%
    \phantomsection{}%
    \ifblank{#2}{%
        \addcontentsline{toc}{section}{Lecture #1}%
        \markboth{Lecture #1}{#3}%
    }{%
        \addcontentsline{toc}{section}{Lecture #1: #2}%
        \markboth{Lecture #1: #2}{#3}%
    }%
}

        {\LARGE\bfseries \coursename}\par
        \vspace{0.4cm}
        {\large \coursenumber\ \ensuremath{\cdot}\ \noteauthor}\par
        \vspace{0.4cm}
        {\large \semester}\par
        \vfill
        {\normalsize \noteauthor}\par
        \vspace*{2cm}
    \end{titlepage}%
}
